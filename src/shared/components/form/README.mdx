# Form Components

These components wrap native form elements and integrate with **react-hook-form** controllers. Labels are localized using `react-i18next`, and validation messages from **Zod** schemas are displayed automatically.

```tsx
import { z } from "zod";
import { zodResolver } from "@hookform/resolvers/zod";
import { FormProvider, useForm } from "react-hook-form";
import { Input, Select, DatePicker, FileUploader } from "@shared/components/form";

const schema = z.object({
  name: z.string().min(1, "Required"),
  country: z.string().min(1, "Required"),
  birthDate: z.string(),
  avatar: z.any().optional(),
});

export function Example() {
  const methods = useForm({ resolver: zodResolver(schema) });
  const onSubmit = methods.handleSubmit((values) => console.log(values));

  return (
    <FormProvider {...methods}>
      <form onSubmit={onSubmit} className="space-y-4">
        <Input name="name" label="form.name" />
        <Select
          name="country"
          label="form.country"
          options={[{ value: "us", label: "United States" }, { value: "ca", label: "Canada" }]}
        />
        <DatePicker name="birthDate" label="form.birthDate" />
        <FileUploader name="avatar" label="form.avatar" />
        <button type="submit">Submit</button>
      </form>
    </FormProvider>
  );
}
```
